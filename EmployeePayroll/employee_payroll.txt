create database payroll_service;
show databases;
use payroll_service;

//UC2

create table employee_payroll
    -> (
    -> id Int unsigned NOT NULL AUTO_INCREMENT,
    -> name Varchar(30) NOT NULL,
    -> salary Double NOT NULL,
    -> start Date NOT NULL,
    -> PRIMARY KEY (id)
    -> );
DESCRIBE employee_payroll;
